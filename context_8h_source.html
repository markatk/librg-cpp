318  template<class T, class P>
319  class Pool {
320  private:
321  std::map<T *, std::shared_ptr<P>> _entries;
322 
323  public:
324  Pool() = default;
325  virtual ~Pool() = default;
326 
327  void add(T *value, const std::shared_ptr<P> &entry) {
328  assert(value != nullptr);
329  assert(entry != nullptr);
330 
331  _entries[value] = entry;
332  }
333 
334  void remove(T *value) {
335  assert(value != nullptr);
336 
337  auto it = _entries.find(value);
338  assert(it != _entries.end());
339 
340  _entries.erase(it);
341  }
342 
343  std::shared_ptr<P> get(T *value) const {
344  assert(value != nullptr);
345 
346  try {
347  return _entries.at(value);
348  } catch (std::out_of_range &) {
349  return nullptr;
350  }
351  }
